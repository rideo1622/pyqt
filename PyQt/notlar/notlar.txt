    Örnek olarak; 
        from PyQt6.QtWidgets import QApplication, QWidget
        import sys


2. app = QApplication(sys.argv) ile uygulama oluşturulur. 


3. Pencere oluşturulması;
    
    windows = QWidget() sayesinde, pencere oluşturulur.


4. uygulama başlatılır;

    sys.exit(app.exec())
